\def\MyIconBookmark#1#2#3{\startpositioning
\position(-2.5,-2){\starttikzpicture[remember picture,overlay]
                  \ifodd\pagenumber 
                    \pgfmathsetmacro\x{-2.1}
                    \pgfmathsetmacro\y{-3}
                  \else
                    \pgfmathsetmacro\x{3.3}
                    \pgfmathsetmacro\y{-3}               
                  \fi
                    \filldraw[#1] (\x,\y) circle [radius=5mm];
                    \node[color=white] at (\x,\y) {\symbol[#2][#3]};
                    \draw[#1,line width=2pt] (\x+0.5,\y) -- (\x+1,\y);
                  \stoptikzpicture}
 \stoppositioning}
