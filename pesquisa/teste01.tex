\newcommand{\watermarkOdd}{teste.pdf} % Arquivo para páginas ímpares
\newcommand{\watermarkEven}{teste2.pdf} % Arquivo para páginas pares
\newcommand{\excludedpage}{1} % Número da página em que você não deseja que o watermark seja exibido

\newcommand{\AddWatermark}{
  \ifnum\value{page}=\excludedpage
    % Não exibir watermark nesta página
  \else
    \begin{tikzpicture}[remember picture,overlay]
      \ifodd\value{page}
        \node at (current page.center) {\includegraphics[width=\paperwidth,height=\paperheight]{\watermarkOdd}};
      \else
        \node at (current page.center) {\includegraphics[width=\paperwidth,height=\paperheight]{\watermarkEven}};
      \fi
    \end{tikzpicture}
  \fi
}

\backgroundsetup{
  contents={\AddWatermark},
  scale=1,
  angle=0,
  opacity=1,
  position=current page.center,
  pages={odd}
}
