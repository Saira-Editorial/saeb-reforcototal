\documentclass{article}
\usepackage{pgfmath}

\begin{document}

% Realizando os cálculos
\pgfmathsetmacro{\firstTerm}{pow(2,2)} % Calcula 2^2 e armazena o resultado em \firstTerm
\pgfmathsetmacro{\secondTerm}{pow(3,2)} % Calcula 3^2 e armazena o resultado em \secondTerm
\pgfmathsetmacro{\result}{\firstTerm + \secondTerm} % Soma \firstTerm e \secondTerm e armazena o resultado em \result

% Exibindo a fórmula e o resultado
$2^2 + 3^2 = \result$

\end{document}
